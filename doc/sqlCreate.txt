Following SQL statements were used to create a postgresql database for storing the recipes


create table recipe_table (
	recipeID BIGSERIAL primary key,
	title varchar(125),
	serves varchar(30)
);

create table ingredient_table (
	ingredientID BIGSERIAL primary key,
	recipeID BIGSERIAL references recipe_table(recipeID),
	ingredient varchar(255) not null,
	quantity varchar(255) not null
);

create table instruction_table (
	instructionID BIGSERIAL primary key,
	recipeID BIGSERIAL references recipe_table(recipeID),
	stepIndex int not null,
	step varchar(255)
);

create table picture_table (
	pictureID BIGSERIAL primary key,
	recipeID BIGSERIAL references recipe_table(recipeID),
	pictureLink varchar(2048)
);

create table category_table (
	categoryID BIGSERIAL primary key,
	recipeID BIGSERIAL references recipe_table(recipeID),
	category varchar(60)
);
